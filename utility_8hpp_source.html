128 template <typename T>
129 [[nodiscard]] inline T convert_to(const std::string_view str) {
130  if constexpr (std::is_same_v<detail::remove_cvref_t<T>, std::string>) {
131  // convert string_view to string
132  return std::string{ trim(str) };
133  } else if constexpr (std::is_same_v<detail::remove_cvref_t<T>, bool>) {
134  const std::string lower_case_str = to_lower_case(trim(str));
135  // the string true
136  if (lower_case_str == "true") {
137  return true;
138  }
139  // the string false
140  if (lower_case_str == "false") {
141  return false;
142  }
143  // convert a number to its "long long" value and convert it to a bool: 0 -> false, otherwise true
144  return static_cast<bool>(convert_to<long long>(str));
145  } else if constexpr (std::is_same_v<detail::remove_cvref_t<T>, char>) {
146  const std::string_view trimmed = trim(str);
147  // since we expect a character, after trimming the string must only contain exactly one character
148  if (trimmed.size() != 1) {
149  throw std::runtime_error{ fmt::format("Can't convert '{}' to a value of type char!", str) };
150  }
151  return trimmed.front();
152  } else if constexpr (std::is_floating_point_v<detail::remove_cvref_t<T>>) {
153  if constexpr (std::is_same_v<detail::remove_cvref_t<T>, float>) {
154  return std::stof(std::string{ str });
155  } else if constexpr (std::is_same_v<detail::remove_cvref_t<T>, double>) {
156  return std::stod(std::string{ str });
157  } else {
158  return std::stold(std::string{ str });
159  }
160  } else {
161  // remove leading whitespaces
162  const std::string_view trimmed_str = trim(str);
163 
164  // convert string to value fo type T
165  T val;
166  auto res = std::from_chars(trimmed_str.data(), trimmed_str.data() + trimmed_str.size(), val);
167  if (res.ec != std::errc{}) {
168  throw std::runtime_error{ fmt::format("Can't convert '{}' to a value of type T!", str) };
169  }
170  return val;
171  }
172 }

182 [[nodiscard]] inline std::vector<T> split_as(const std::string_view str, const char delim = ' ') {
183  std::vector<std::string> split_str;
184 
185  // if the input str is empty, return an empty vector
186  if (str.empty()) {
187  return split_str;
188  }
189 
190  std::string_view::size_type pos = 0;
191  std::string_view::size_type next = 0;
192  while (next != std::string_view::npos) {
193  next = str.find_first_of(delim, pos);
194  split_str.emplace_back(convert_to<T>(next == std::string_view::npos ? str.substr(pos) : str.substr(pos, next - pos)));
195  pos = next + 1;
196  }
